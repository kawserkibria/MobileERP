
SELECT ACC_COMPANY_VOUCHER.COMP_REF_NO, ACC_COMPANY_VOUCHER.COMP_VOUCHER_NET_AMOUNT,SUM(ACC_VOUCHER.VOUCHER_CREDIT_AMOUNT) 
FROM ACC_COMPANY_VOUCHER ,ACC_VOUCHER WHERE ACC_COMPANY_VOUCHER.COMP_REF_NO =ACC_VOUCHER.COMP_REF_NO AND  ACC_VOUCHER.COMP_VOUCHER_DATE 
BETWEEN  Convert (DateTime  ,'01/07/2020', 103)  AND Convert (DateTime  ,'31/07/2020', 103)  AND ACC_VOUCHER.COMP_VOUCHER_TYPE =1 
--AND ACC_VOUCHER.COMP_REF_NO ='RV0001RV#020402'
GROUP BY ACC_COMPANY_VOUCHER.COMP_REF_NO,ACC_COMPANY_VOUCHER.COMP_VOUCHER_NET_AMOUNT
HAVING  (ACC_COMPANY_VOUCHER.COMP_VOUCHER_NET_AMOUNT) <> SUM(ACC_VOUCHER.VOUCHER_CREDIT_AMOUNT)
